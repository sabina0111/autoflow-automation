# Quick Deployment Commands
# Copy and paste these commands one by one in PowerShell

# 1. Initialize Git (if not already done)
git init

# 2. Add all files
git add .

# 3. Commit
git commit -m "Initial commit - AutoFlow Workflow Automation System"

# 4. Create GitHub repo at: https://github.com/new
#    Then copy the commands GitHub gives you, OR use these:
#    (Replace YOUR_USERNAME and REPO_NAME with your actual values)

# 5. Connect to GitHub
git remote add origin https://github.com/YOUR_USERNAME/REPO_NAME.git

# 6. Push to GitHub
git branch -M main
git push -u origin main

# Done! Now go to vercel.com to deploy
